	BoldFont=rb.ttf,
	ItalicFont=ri.ttf,
	BoldItalicFont=rbi.ttf

	Path = f/,
	BoldFont=rbb.ttf,
	ItalicFont=rli.ttf,
	BoldItalicFont=rbbi.ttf
		]{rl.ttf}
